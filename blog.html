## require packages
require(pacman)
p_load(leaflet,sf,osmdata)

## getting data
bus_station = opq(bbox = getbb("Bogotá Colombia")) %>%
              add_osm_feature(key = "amenity", value = "bus_station") %>%
              osmdata_sf() %>% .$osm_points

## view data
leaflet() %>% addTiles() %>% 
addCircleMarkers(data=bus_station , weight=1 , col="red")
